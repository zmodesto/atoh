----------------------------------------------------------------------

-- [to unstage all files that have been added but not committed]
    > git reset


-- [to add all files except those specific by gitignore]
    > git add .

-- [to commit all added files and include a commit message]
    > git commit -am "words words words"

-- [to push commits from origin to master (typically local to github)]
    > git push origin master
